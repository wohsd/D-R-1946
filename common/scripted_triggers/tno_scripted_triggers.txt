can_claim_antarctica = {
	NOT = { has_game_rule = { rule = antarctica_content option = BLOCKED } }
	NOT = { has_global_flag = ZZZ_isnt_ai }
	OR = {
		AND = { #Generic countries
			NOT = { has_game_rule = { rule = antarctica_content option = LIMITED } }
			is_major = yes
			any_state = {
				is_owned_by = ROOT
				is_coastal = yes # Don't allow majors without sea access
			}
			has_navy_size = { #Only allow countries that have a large navy
				size > 3
				type = capital_ship
			}
			is_ai = no #Don't want AI Sweden claiming Antarctica
			has_capitulated = no #?????
		}
		#OR = {
		historical_antarctic_claimants = yes
		#}
	}
}
