add_namespace = tno_news

#News

#Argentina wins the South Orkney War
news_event = {
	id = tno_news.1
	title = tno_news.1.t
	desc = tno_news.1.desc
	picture = GFX_news_event_generic_south_american_dreadnought_race
	#fire_only_once = yes
	is_triggered_only = yes
	major = yes

	option = { #Option for Britain
		name = tno_news.1.a
		trigger = {
			original_tag = ENG
		}
		add_stability = -0.05
		add_war_support = -0.05
		add_political_power = -50
	}
	option = { #Option for Argentina
		name = tno_news.1.b
		trigger = {
			original_tag = ARG
		}
		add_stability = 0.05
		add_war_support = 0.05
		add_political_power = 50
	}
	option = { #Option for other countries
		name = tno_news.1.c
		trigger = {
			NOT = {
				original_tag = ENG
				original_tag = ARG
			}
		}
	}
}
